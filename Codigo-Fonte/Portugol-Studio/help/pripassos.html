Exemplo 3

Programa que pede ao usuario 2 valores e Escreve os Valores digitados na tela
Obs: "\n" é um comando para pular linha 



programa 
   {
     funcao inicio (cadeia parametros[])
       {

        inteiro a, b 
        escreva ("Digite o Valor de A ")
        leia (a)
        escreva ("\nDigite o Valor de B ")
        leia (b)
      
        escreva ("\nOs valores digitados foram ",a," e ",b)
 
      }
 }
